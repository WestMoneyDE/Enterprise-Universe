    constructor() {
        this.apis = new Map();
        this.cache = new Map();
    }

    registerAPI(name, config) {
        this.apis.set(name, {
            baseUrl: config.baseUrl,
            auth: config.auth,
            authType: config.authType, // 'apiKey', 'OAuth', 'Bearer', 'none'
            rateLimit: config.rateLimit || 100,
            endpoints: config.endpoints || {}
        });
    }

    async call(apiName, endpoint, params = {}) {
        const api = this.apis.get(apiName);
        if (!api) throw new Error(`API ${apiName} not registered`);

        const headers = this.buildHeaders(api);
        const url = this.buildUrl(api.baseUrl, endpoint, params);

        const response = await fetch(url, { headers });
        return response.json();
    }

    buildHeaders(api) {
        const headers = { 'Content-Type': 'application/json' };
        
        switch (api.authType) {
            case 'apiKey':
                headers['X-API-Key'] = api.auth;
                break;
            case 'Bearer':
                headers['Authorization'] = `Bearer ${api.auth}`;
                break;
            case 'OAuth':
                headers['Authorization'] = `OAuth ${api.auth}`;
                break;
        }
        
        return headers;
    }
}
