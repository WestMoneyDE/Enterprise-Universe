## ğŸ“Š DATABASE SCHEMA

```sql
-- Core Tables
CREATE TABLE connectors (
    id UUID PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    type VARCHAR(50) NOT NULL, -- 'erp', 'crm', 'database', 'automation'
    provider VARCHAR(50) NOT NULL, -- 'odoo', 'erpnext', 'hubspot', etc.
    config JSONB NOT NULL,
    status VARCHAR(20) DEFAULT 'inactive',
    last_sync TIMESTAMP,
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE sync_mappings (
    id UUID PRIMARY KEY,
    connector_id UUID REFERENCES connectors(id),
    source_entity VARCHAR(100),
    target_entity VARCHAR(100),
    field_mappings JSONB,
    sync_direction VARCHAR(20), -- 'bidirectional', 'source_to_target', 'target_to_source'
    enabled BOOLEAN DEFAULT true,
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE sync_logs (
    id UUID PRIMARY KEY,
    connector_id UUID REFERENCES connectors(id),
    entity_type VARCHAR(100),
    operation VARCHAR(20), -- 'create', 'update', 'delete'
    source_id VARCHAR(255),
    target_id VARCHAR(255),
    status VARCHAR(20),
    error_message TEXT,
    executed_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE unified_contacts (
    id UUID PRIMARY KEY,
    source_connector UUID REFERENCES connectors(id),
    source_id VARCHAR(255),
    first_name VARCHAR(100),
    last_name VARCHAR(100),
    email VARCHAR(255),
    phone VARCHAR(50),
    company VARCHAR(255),
    job_title VARCHAR(100),
    address JSONB,
    type VARCHAR(50),
    status VARCHAR(50),
    consent JSONB,
    custom_fields JSONB,
    hash VARCHAR(32), -- MD5 for change detection
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

-- Indexes
CREATE INDEX idx_contacts_email ON unified_contacts(email);
CREATE INDEX idx_contacts_source ON unified_contacts(source_connector, source_id);
CREATE INDEX idx_sync_logs_connector ON sync_logs(connector_id, executed_at);
```
