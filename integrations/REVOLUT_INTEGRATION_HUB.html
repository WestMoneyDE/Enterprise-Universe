Access Token abrufen

javascript 
const jwt = require('jsonwebtoken');

const getAccessToken = async (authCode) => {
  const clientAssertion = jwt.sign({
    iss: REVOLUT_CLIENT_ID,
    sub: REVOLUT_CLIENT_ID,
    aud: 'https://revolut.com',
    exp: Math.floor(Date.now() / 1000) + 120
  }, privateKey, { algorithm: 'RS256' });

  const response = await fetch(
    'https://b2b.revolut.com/api/1.0/auth/token', {
    method: 'POST',
    headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
    body: new URLSearchParams({
      grant_type: 'authorization_code',
      code: authCode,
      client_assertion_type: 'urn:ietf:params:oauth:client-assertion-type:jwt-bearer',
      client_assertion: clientAssertion
    })
  });

  return response.json();
};

TypeScript
typescript 
// services/revolut.service.ts

export class RevolutService {
  private baseUrl = 'https://b2b.revolut.com/api/1.0';
  private accessToken: string;

  async getAccounts() {
    return this.request('/accounts');
  }

  async getBalance(accountId: string) {
    return this.request(`/accounts/${accountId}`);
  }

  async createPayment(data: {
    accountId: string;
    counterpartyId: string;
    amount: number;
    currency: string;
    reference: string;
  }) {
    return this.request('/pay', {
      method: 'POST',
      body: JSON.stringify({
        request_id: `wmo_${Date.now()}`,
        account_id: data.accountId,
        receiver: { counterparty_id: data.counterpartyId },
        amount: data.amount,
        currency: data.currency,
        reference: data.reference
      })
    });
  }

  async getTransactions(params?: { 
    from?: string; 
    to?: string; 
    count?: number 
  }) {
    const query = new URLSearchParams(params);
    return this.request(`/transactions?${query}`);
  }

  private async request(endpoint: string, options = {}) {
    const response = await fetch(`${this.baseUrl}${endpoint}`, {
      ...options,
      headers: {
        'Authorization': `Bearer ${this.accessToken}`,
        'Content-Type': 'application/json'
      }
    });
    return response.json();
  }
}
